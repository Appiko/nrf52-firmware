74 void ms_timer_start(ms_timer_num id, ms_timer_mode mode, uint32_t ticks, void (*handler)(void))
75 {
76 
77  /* make sure the number of ticks to interrupt is less than 2^24 */
78  ticks = RTC_WRAP(ticks);
79 
80  ms_timer[id].timer_handler = handler;
81  if (mode == MS_SINGLE_CALL)
82  {
83  ms_timer[id].timer_mode = MS_SINGLE_CALL;
84  }
85  else
86  {
87  ms_timer[id].timer_mode = ticks;
88  }
89 
90  RTC_ID->CC[id] = RTC_ID->COUNTER + ticks;
91 
92  RTC_ID->EVENTS_COMPARE[id] = 0;
93  RTC_ID->EVTENSET = 1 << (RTC_INTENSET_COMPARE0_Pos + id);
94  RTC_ID->INTENSET = 1 << (RTC_INTENSET_COMPARE0_Pos + id);
95 
96  if (ms_timers_status == 0)
97  {
98  RTC_ID->TASKS_START = 1;
99  }
100  ms_timers_status |= 1 << id;
101 }

103 void ms_timer_stop(ms_timer_num id)
104 {
105  ms_timer[id].timer_mode = MS_SINGLE_CALL;
106  ms_timers_status &= ~(1 << id);
107  RTC_ID->EVTENCLR = 1 << (RTC_INTENSET_COMPARE0_Pos + id);
108  RTC_ID->INTENCLR = 1 << (RTC_INTENSET_COMPARE0_Pos + id);
109 
110  if (ms_timers_status == 0)
111  {
112  RTC_ID->TASKS_STOP = 1;
113  }
114 }
