38 void lfclk_init(nrf_clock_lfclk_t lfclk_src)
39 {
40  if (nrf_clock_lf_is_running() == false)
41  {
42  nrf_clock_lf_src_set(lfclk_src);
43  nrf_clock_int_enable(NRF_CLOCK_INT_LF_STARTED_MASK);
44  NVIC_ClearPendingIRQ(POWER_CLOCK_IRQn);
45 
46  // Enable wake-up on any event or interrupt (even disabled)
47  SCB->SCR |= SCB_SCR_SEVONPEND_Msk;
48 
49  nrf_clock_event_clear(NRF_CLOCK_EVENT_LFCLKSTARTED);
50  nrf_clock_task_trigger(NRF_CLOCK_TASK_LFCLKSTART);
51  /* Wait for the external oscillator to start up. */
52  while (nrf_clock_event_check(NRF_CLOCK_EVENT_LFCLKSTARTED) == 0)
53  {
54  __WFE();
55  }
56  /* Clear the event and the pending interrupt */
57  nrf_clock_event_clear(NRF_CLOCK_EVENT_LFCLKSTARTED);
58  NVIC_ClearPendingIRQ(POWER_CLOCK_IRQn);
59  nrf_clock_int_disable(NRF_CLOCK_INT_LF_STARTED_MASK);
60 
61  // Enable wake-up on only enabled interrupts
62  SCB->SCR &= (~(SCB_SCR_SEVONPEND_Msk));
63 
64 #ifdef NRF52832
65  //Due to errata 20 in Eng rev 1
66  NRF_RTC0->TASKS_STOP = 0;
67  NRF_RTC1->TASKS_STOP = 0;
68  NRF_RTC2->TASKS_STOP = 0;
69 #endif
70  }
71 }
